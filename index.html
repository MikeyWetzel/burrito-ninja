<!DOCTYPE html> <html lang="en"> <head> <meta charset="UTF-8"> <meta name="viewport" content="width=device-width, initial-scale=1.0"> <title>Burrito Ninja</title> <style> * { margin: 0; padding: 0; box-sizing: border-box; } body { background: linear-gradient(to top, #87CEEB 0%, #98D8E8 100%); overflow: hidden; font-family: Arial, sans-serif; } canvas { display: block; width: 100vw; height: 100vh; touch-action: none; } </style> </head> <body> <canvas id="gameCanvas"></canvas> <script> const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); let score = 0; let burritos = []; let trail = []; let mouse = { x: 0, y: 0, down: false, prevX: 0, prevY: 0 }; function resizeCanvas() { canvas.width = window.innerWidth; canvas.height = window.innerHeight; } window.addEventListener('resize', resizeCanvas); resizeCanvas(); // Mouse events canvas.addEventListener('mousemove', (e) => { mouse.prevX = mouse.x; mouse.prevY = mouse.y; mouse.x = e.clientX; mouse.y = e.clientY; }); canvas.addEventListener('mousedown', () => mouse.down = true); canvas.addEventListener('mouseup', () => { mouse.down = false; trail = []; }); // Touch events canvas.addEventListener('touchstart', (e) => { e.preventDefault(); mouse.down = true; }, { passive: false }); canvas.addEventListener('touchmove', (e) => { e.preventDefault(); const touch = e.touches[0]; mouse.prevX = mouse.x; mouse.prevY = mouse.y; mouse.x = touch.clientX; mouse.y = touch.clientY; }, { passive: false }); canvas.addEventListener('touchend', (e) => { e.preventDefault(); mouse.down = false; trail = []; }, { passive: false }); function spawnBurrito() { if (Math.random() < 0.015) { burritos.push({ x: Math.random() * canvas.width, y: canvas.height + 50, vx: (Math.random() - 0.5) * 200, vy: -250 - Math.random() * 150, r: 25 + Math.random() * 25, sliced: false, life: 1.0 }); } } function lineIntersectsCircle(x1, y1, x2, y2, cx, cy, r) { const dx = x2 - x1; const dy = y2 - y1; const lenSq = dx * dx + dy * dy; if (lenSq === 0) { return Math.hypot(cx - x1, cy - y1) < r; } let t = ((cx - x1) * dx + (cy - y1) * dy) / lenSq; t = Math.max(0, Math.min(1, t)); const projX = x1 + t * dx; const projY = y1 + t * dy; return Math.hypot(cx - projX, cy - projY) < r; } function update(dt) { spawnBurrito(); // Update trail if (mouse.down) { trail.push({ x: mouse.x, y: mouse.y }); if (trail.length > 20) trail.shift(); } // Update burritos for (let i = burritos.length - 1; i >= 0; i--) { const b = burritos[i]; b.x += b.vx * dt; b.y += b.vy * dt; if (!b.sliced) { // Check slice if (mouse.down && trail.length > 1) { // Check last segment const last = trail[trail.length - 2]; if (lineIntersectsCircle(last.x, last.y, mouse.x, mouse.y, b.x, b.y, b.r)) { b.sliced = true; score += 10; } } // Remove offscreen if (b.y + b.r < -50 || b.x + b.r < -50 || b.x - b.r > canvas.width + 50) { burritos.splice(i, 1); continue; } } else { // Sliced: explode b.vy += 500 * dt; // gravity b.vx += (Math.random() - 0.5) * 400 * dt; b.life -= dt * 3; b.r *= 0.98; if (b.life <= 0 || b.r < 2 || b.y > canvas.height + 100) { burritos.splice(i, 1); } } } } function draw() { // Sky bg const gradient = ctx.createLinearGradient(0, 0, 0, canvas.height); gradient.addColorStop(0, '#87CEEB'); gradient.addColorStop(1, '#E0F6FF'); ctx.fillStyle = gradient; ctx.fillRect(0, 0, canvas.width, canvas.height); // Burritos burritos.forEach(b => { ctx.save(); ctx.translate(b.x, b.y); if (b.sliced) { // Sliced halves ctx.fillStyle = '#A0522D'; // sliced brown ctx.shadowColor = 'rgba(0,0,0,0.5)'; ctx.shadowBlur = 10; // Left half ctx.beginPath(); ctx.arc(-b.r / 3, -b.r / 3, b.r * 0.6, 0, Math.PI * 2); ctx.fill(); // Right half ctx.beginPath(); ctx.arc(b.r / 3, b.r / 3, b.r * 0.6, 0, Math.PI * 2); ctx.fill(); ctx.shadowBlur = 0; } else { // Intact burrito ctx.fillStyle = '#D2691E'; // tortilla ctx.beginPath(); ctx.arc(0, 0, b.r, 0, Math.PI * 2); ctx.fill(); // Fillings ctx.fillStyle = '#228B22'; // lettuce ctx.fillRect(-b.r / 2, -b.r / 3, b.r / 2, b.r / 3); ctx.fillStyle = '#FF6347'; // salsa ctx.beginPath(); ctx.arc(b.r / 3, 0, b.r / 4, 0, Math.PI * 2); ctx.fill(); ctx.fillStyle = '#FFD700'; // cheese ctx.fillRect(-b.r / 3, b.r / 6, b.r / 1.5, b.r / 6); } ctx.restore(); }); // Sword trail if (trail.length > 1) { ctx.strokeStyle = 'rgba(255, 255, 255, 0.6)'; ctx.lineWidth = 8; ctx.lineCap = 'round'; ctx.lineJoin = 'round'; ctx.shadowColor = 'white'; ctx.shadowBlur = 15; ctx.beginPath(); trail.forEach((p, i) => { if (i === 0) ctx.moveTo(p.x, p.y); else ctx.lineTo(p.x, p.y); }); ctx.stroke(); ctx.shadowBlur = 0; } // Score ctx.fillStyle = 'white'; ctx.shadowColor = 'black'; ctx.shadowBlur = 4; ctx.font = `${Math.min(48, canvas.height / 20)}px Arial Black, sans-serif`; ctx.textAlign = 'left'; ctx.textBaseline = 'top'; ctx.fillText(`Score: ${score}`, 20, 20); ctx.shadowBlur = 0; } let lastTime = 0; function gameLoop(currentTime) { const dt = Math.min((currentTime - lastTime) / 1000, 0.05); // cap dt lastTime = currentTime; update(dt); draw(); requestAnimationFrame(gameLoop); } requestAnimationFrame(gameLoop); </script> </body> </html>
